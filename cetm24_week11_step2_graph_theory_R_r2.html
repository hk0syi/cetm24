library(igraph)

# ---- Function to build graph, analyze, and plot ----
analyze_bridges <- function(bridges) {
  cat("Bridges list:\n")
  print(bridges)
  
  # Build igraph object
  g <- graph_from_data_frame(bridges, directed = FALSE)
  E(g)$name <- bridges$name
  
  # ----- Adjacency matrix -----
  adj_matrix <- as.matrix(as_adjacency_matrix(g, sparse = FALSE))
  cat("\nAdjacency Matrix:\n")
  print(adj_matrix)
  
  # ----- Build Incidence matrix manually -----
  vertices <- V(g)$name
  edges <- E(g)
  inc_matrix <- matrix(0, nrow=length(edges), ncol=length(vertices),
                       dimnames=list(E(g)$name, vertices))
  
  for (i in seq_along(edges)) {
    v_pair <- ends(g, edges[i])
    inc_matrix[i, v_pair] <- 1
  }
  
  cat("\nIncidence Matrix (rows=bridges, cols=landmasses):\n")
  print(inc_matrix)
  
  # ----- Degrees -----
  deg <- degree(g)
  cat("\nVertex Degrees:\n")
  print(deg)
  
  # ----- Eulerian check -----
  odd_vertices <- sum(deg %% 2 == 1)
  if (odd_vertices == 0) {
    cat("\nEulerian circuit exists (closed walk possible).\n")
  } else if (odd_vertices == 2) {
    cat("\nEulerian path exists (open walk possible).\n")
  } else {
    cat("\nNo Eulerian walk exists (", odd_vertices, " odd-degree vertices).\n")
  }
  
  # ----- Graph Plot -----
  plot(g,
       layout = layout_with_fr,
       vertex.size = 30,
       vertex.color = "lightblue",
       vertex.label.color = "black",
       edge.width = 2,
       edge.label = E(g)$name,
       edge.label.color = "red")
}

library(igraph)

# ----- Helper: find Eulerian path using Hierholzers algorithm -----
find_eulerian_path <- function(g) {
  deg <- degree(g)
  odd_vertices <- V(g)[deg %% 2 == 1]
  
  # start: if Eulerian path, must start on odd vertex, else pick any
  if (length(odd_vertices) == 0) {
    start <- V(g)[1]
  } else if (length(odd_vertices) == 2) {
    start <- odd_vertices[1]
  } else {
    stop("Graph is not Eulerian, can't find Eulerian path")
  }
  
  path <- c()
  stack <- c(as_ids(start))
  g_copy <- g
  
  while (length(stack) > 0) {
    v <- tail(stack, 1)
    if (degree(g_copy, v) > 0) {
      # pick an edge
      e <- E(g_copy)[incident(g_copy, v)[1]]
      nbr <- ends(g_copy, e)[1:2]
      next_v <- ifelse(nbr[1] == v, nbr[2], nbr[1])
      stack <- c(stack, next_v)
      g_copy <- delete_edges(g_copy, e)  # remove used edge
    } else {
      path <- c(path, v)
      stack <- stack[-length(stack)]
    }
  }
  
  return(rev(path)) # sequence of vertices
}

# ----- Analyzer function -----
analyze_bridges <- function(bridges) {
  g <- graph_from_data_frame(bridges, directed = FALSE)
  E(g)$name <- bridges$name
  
  deg <- degree(g)
  odd_vertices <- sum(deg %% 2 == 1)
  
  cat("\nDegrees:\n"); print(deg)
  
  if (odd_vertices == 0) {
    cat("Eulerian circuit exists.\n")
    path <- find_eulerian_path(g)
    cat("Eulerian circuit (vertices):", paste(path, collapse=" -> "), "\n")
  } else if (odd_vertices == 2) {
    cat("Eulerian path exists.\n")
    path <- find_eulerian_path(g)
    cat("Eulerian path (vertices):", paste(path, collapse=" -> "), "\n")
  } else {
    cat("No Eulerian walk exists.\n")
  }
  
  plot(g,
       layout = layout_with_fr,
       vertex.size=30,
       vertex.color="lightblue",
       vertex.label.color="black",
       edge.width=2,
       edge.label=E(g)$name,
       edge.label.color="red")
}
