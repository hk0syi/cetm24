# 1) Heatmap for a selected bicluster
plot_cc_heatmap <- function(data_matrix, biclust_result, bicluster_num = 1) {
  if (is.null(biclust_result) || biclust_result@Number == 0) {
    cat("No CC biclusters to plot.\n"); return(invisible(NULL))
  }
  if (bicluster_num > biclust_result@Number) {
    cat("Requested bicluster exceeds available count.\n"); return(invisible(NULL))
  }
  rows <- which(biclust_result@RowxNumber[, bicluster_num])
  cols <- which(biclust_result@NumberxCol[bicluster_num, ])
  cat(sprintf("CC Bicluster %d: %d rows x %d cols\n", bicluster_num, length(rows), length(cols)))
  if (length(rows) < 2 || length(cols) < 2) {
    cat("Insufficient size for heatmap (need >=2 rows and >=2 cols).\n"); return(invisible(NULL))
  }
  subset_data <- data_matrix[rows, cols, drop = FALSE]
  heatmap(subset_data,
          main = paste("CC Bicluster", bicluster_num, "Heatmap"),
          xlab = "Features", ylab = "Customers",
          col = colorRampPalette(c("blue", "white", "red"))(50),
          scale = "none")
}

Rows: customers in bicluster 5 (row labels at right).

Columns: selected features (looks like Age, Annual Income (k), and Spending Score).

Colors are z-scores: red = above-feature mean, blue = below-feature mean, white = near mean.

# 2) Bicluster size bar plot
plot_cc_sizes <- function(biclust_result) {
  if (is.null(biclust_result) || biclust_result@Number == 0) {
    cat("No CC biclusters to plot.\n"); return(invisible(NULL))
  }
  sizes <- sapply(1:biclust_result@Number, function(i) sum(biclust_result@RowxNumber[, i]))
  barplot(sizes, names.arg = paste0("BC", 1:length(sizes)),
          main = "CC Bicluster Sizes (rows/customers)",
          xlab = "Bicluster", ylab = "Number of Customers",
          col = "steelblue")
}

# 3) Simple scatter plots with bicluster membership highlighting (requires ggplot2 and original data_clean)
#    Provide the column names for Age, Income, and Spending if different.
plot_cc_scatter <- function(data_clean, biclust_result,
                            age_col = NULL, income_col = NULL, spending_col = NULL) {
  if (is.null(biclust_result) || biclust_result@Number == 0) {
    cat("No CC biclusters to plot.\n"); return(invisible(NULL))
  }
  # Guess columns if not provided
  if (is.null(age_col))      age_col      <- names(data_clean)[grepl("age", names(data_clean), ignore.case = TRUE)][1]
  if (is.null(income_col))   income_col   <- names(data_clean)[grepl("income", names(data_clean), ignore.case = TRUE)][1]
  if (is.null(spending_col)) spending_col <- names(data_clean)[grepl("spending|score", names(data_clean), ignore.case = TRUE)][1]
  # Fallbacks if guessing fails
  if (is.na(age_col))      age_col      <- names(data_clean)[1]
  if (is.na(income_col))   income_col   <- names(data_clean)[2]
  if (is.na(spending_col)) spending_col <- names(data_clean)[3]

  plot_df <- data_clean
  plot_df$Bicluster <- "None"
  for (i in 1:min(biclust_result@Number, 5)) {
    rows <- which(biclust_result@RowxNumber[, i])
    plot_df$Bicluster[rows] <- paste("Bicluster", i)
  }
  plot_df$Bicluster <- as.factor(plot_df$Bicluster)

  p1 <- ggplot(plot_df, aes_string(x = age_col, y = income_col, color = "Bicluster")) +
    geom_point(size = 2, alpha = 0.8) +
    theme_minimal() +
    labs(title = "Age vs Income by CC Bicluster", x = "Age", y = "Annual Income")

  p2 <- ggplot(plot_df, aes_string(x = income_col, y = spending_col, color = "Bicluster")) +
    geom_point(size = 2, alpha = 0.8) +
    theme_minimal() +
    labs(title = "Income vs Spending Score by CC Bicluster", x = "Annual Income", y = "Spending Score")

  print(p1); print(p2)
}

  membership_row <- as.data.frame(biclust_cc@RowxNumber)
  colnames(membership_row) <- paste0("Bicluster_", seq_len(ncol(membership_row)))
  
  # Align CustomerID (handle possible row filtering after scaling)
  # If data_clean rows match a subset of original data, align IDs safely
  if ("CustomerID" %in% names(data)) {
    # If lengths differ, take the first nrow(data_clean) IDs
    customer_ids <- data$CustomerID[seq_len(nrow(data_clean))]
    membership_row <- cbind(CustomerID = customer_ids, membership_row)
  } else {
    membership_row <- cbind(RowIndex = seq_len(nrow(data_clean)), membership_row)
  }
  
  # 2) Column-wise membership (features x biclusters)
  membership_col <- as.data.frame(t(biclust_cc@NumberxCol))
  rownames(membership_col) <- colnames(data_scaled)
  colnames(membership_col) <- paste0("Bicluster_", seq_len(ncol(membership_col)))
  membership_col$Feature <- rownames(membership_col)
  membership_col <- membership_col[, c(ncol(membership_col), 1:(ncol(membership_col)-1))]
  
  # 3) Simple single-label assignments (first bicluster hit, else None)
  bic_labels <- rep("None", nrow(data_clean))
  for (j in seq_len(biclust_cc@Number)) {
    rows_j <- which(biclust_cc@RowxNumber[, j])
    # Only set label if still None (keeps first bicluster assignment)
    to_set <- which(bic_labels[rows_j] == "None")

    CustomerID = if ("CustomerID" %in% names(data)) data$CustomerID[seq_len(nrow(data_clean))] else seq_len(nrow(data_clean)),
    Bicluster = bic_labels,
    stringsAsFactors = FALSE
  )
